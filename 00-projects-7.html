CSS reference

/* ID selectors */ #logo { width: 100px; }
/* Attribute selectors */ [hreflang] { font-style: italic; }
*CSS Properties* /* Color properties */ color: #333; background-color: #f2f2f2;
/* Font properties */ font-size: 16px; font-family: Arial, sans-serif; font-weight: bold; font-style: italic;
/* Box model properties */ width: 100px; height: 100px; padding: 10px; border: 1px solid #333; margin: 10px;
/* Layout properties */ display: block; position: relative; top: 10px; left: 10px;
*CSS Units* /* Absolute units */ width: 100px; height: 100px;
/* Relative units */ font-size: 1.5em; width: 50%;
*CSS Box Model* /* Box sizing */ box-sizing: border-box;
/* Border properties */ border-width: 1px; border-style: solid; border-color: #333;
/* Margin and padding properties */ margin: 10px; padding: 10px;
*CSS Layout* /* Display properties */ display: block; display: inline; display: inline-block;
/* Position properties */ position: static; position: relative; position: absolute; position: fixed;
/* Float properties */ float: left; float: right;
